
// cria repositório local
git init 

//inicia o repositório local
git add .

//salva as alterações do git , ignora mensagens no terminal e deia 
git commit -q -m "Primeiro commit"

//conecta o repositório local a um do github
git remote add origin https://github.com/usuario/nome-repo.git

// cria uma partição dentro do repositório e renomeia ele para "main"(isolamento dentro do código)
git branch -M main



// Manda todas as alterações para a partição "main"
// -u vinculo entre a branch local, fazendo com que em proximos comandos voce não precise especificar o branch no futuro 
// origin nome do repositório 
// main é a branch onde vai ocorrer 
git push -u origin main
